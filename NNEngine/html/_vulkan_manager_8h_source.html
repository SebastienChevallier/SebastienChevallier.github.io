669 template<> struct hash<glm::vec2> {
670 size_t operator()(glm::vec2 const& v) const noexcept {
671 return hash<float>()(v.x) ^ (hash<float>()(v.y) << 1);
672 }
673 };
674 template<> struct hash<glm::vec3> {
675 size_t operator()(glm::vec3 const& v) const noexcept {
676 size_t h1 = hash<float>()(v.x);
677 size_t h2 = hash<float>()(v.y);
678 size_t h3 = hash<float>()(v.z);
679 return ((h1 ^ (h2 << 1)) >> 1) ^ (h3 << 1);
680 }
681 };
682 template<> struct hash<NNE::Systems::Vertex> {
683 size_t operator()(NNE::Systems::Vertex const& vertex) const {
684 return ((hash<glm::vec3>()(vertex.pos) ^
685 (hash<glm::vec3>()(vertex.color) << 1)) >> 1) ^
686 (hash<glm::vec2>()(vertex.texCoord) << 1);
687 }
688 };
