86void NNE::Systems::Application::Update()
87{
88 auto* manager = NNE::Systems::SystemManager::GetInstance();
89 while (!glfwWindowShouldClose(VKManager->window)) {
90 delta = GetDeltaTime();
91 float dtMs = delta * 1000.0f;
92 static float smooth = dtMs;
93 smooth = 0.9f * smooth + 0.1f * dtMs;
94 g_FrameTimeMs = smooth;
95 g_FPS = 1000.0f / std::max(0.001f, g_FrameTimeMs);
96 glfwPollEvents();
97
98 glfwGetWindowSize(VKManager->window, &WIDTH,&HEIGHT);
99
100 manager->UpdateAll(delta);
101
102 for (NNE::AEntity* entity : _entities)
103 {
104 entity->Update(delta);
105 }
106
107 manager->LateUpdateAll(delta);
108
109 for (NNE::AEntity* entity : _entities)
110 {
111 entity->LateUpdate(delta);
112 }
113 }
114 vkDeviceWaitIdle(VKManager->device);

158 auto currentFrame = std::chrono::high_resolution_clock::now();
