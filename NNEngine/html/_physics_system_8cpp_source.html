254bool PhysicsSystem::Raycast(glm::vec3 origin, glm::vec3 direction, float distance,
255 RaycastHit &outHit, JPH::ObjectLayer rayLayer) {
256 auto* system = NNE::Systems::PhysicsSystem::GetInstance();
257 if (!system)
258 return false;
259 JPH::PhysicsSystem& phys = system->physicsSystem;
260
261 class RaycastLayerFilter final : public JPH::ObjectLayerFilter {
262 public:
263 explicit RaycastLayerFilter(JPH::ObjectLayer layer) : mLayer(layer) {}
264 bool ShouldCollide(JPH::ObjectLayer other) const override {
265 auto* phys = NNE::Systems::PhysicsSystem::GetInstance();
266 return phys->LayersShouldCollide(mLayer, other);
267 }
268 private:
269 JPH::ObjectLayer mLayer;
270 };
271
272 JPH::RRayCast ray({ origin.x, origin.y, origin.z },
273 JPH::Vec3(direction.x, direction.y, direction.z) * distance);
274 JPH::RayCastResult result;
275 JPH::BroadPhaseLayerFilter broadPhaseFilter;
276 RaycastLayerFilter objectLayerFilter(rayLayer);
277 JPH::BodyFilter bodyFilter;
278 if (!phys.GetNarrowPhaseQuery().CastRay(
279 ray, result, broadPhaseFilter, objectLayerFilter, bodyFilter))
280 return false;
281
282 outHit.bodyID = result.mBodyID;
283 outHit.subShapeID = result.mSubShapeID2;
284 outHit.fraction = result.mFraction;
285 if (auto* col = system->GetCollider(result.mBodyID))
286 outHit.entity = col->GetEntity();
287 else
288 outHit.entity = nullptr;
289
290 JPH::RVec3 point = ray.GetPointOnRay(result.mFraction);
291 outHit.position = glm::vec3(point.GetX(), point.GetY(), point.GetZ());
292
293 JPH::BodyLockRead lock(phys.GetBodyLockInterface(), result.mBodyID);
294 if (lock.Succeeded()) {
295 const JPH::Body& body = lock.GetBody();
296 JPH::Vec3 normal = body.GetWorldSpaceSurfaceNormal(result.mSubShapeID2, point);
297 outHit.normal = glm::vec3(normal.GetX(), normal.GetY(), normal.GetZ());
298 }
299 return true;
300}

307void PhysicsSystem::ContactListenerImpl::OnContactAdded(
308 const JPH::Body &body1, const JPH::Body &body2,
309 const JPH::ContactManifold &manifold, JPH::ContactSettings &) {
310 auto *colliderA =
311 NNE::Systems::PhysicsSystem::GetInstance()->GetCollider(body1.GetID());
312 auto *colliderB =
313 NNE::Systems::PhysicsSystem::GetInstance()->GetCollider(body2.GetID());
314
315 if (colliderA && colliderB) {
316 if (colliderA->IsTrigger() || colliderB->IsTrigger()) {
317 if (colliderA->IsTrigger())
318 colliderA->OnTriggerHit(colliderB);
319 else
320 colliderA->OnHit(colliderB);
321 if (colliderB->IsTrigger())
322 colliderB->OnTriggerHit(colliderA);
323 else
324 colliderB->OnHit(colliderA);
325 } else {
326 colliderA->OnHit(colliderB);
327 colliderB->OnHit(colliderA);
328 }
329 }
330}
