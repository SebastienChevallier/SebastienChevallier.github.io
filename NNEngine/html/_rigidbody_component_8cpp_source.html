19void RigidbodyComponent::Awake() {

21 auto* transform = GetEntity()->GetComponent<NNE::Component::TransformComponent>();
22 auto* collider = GetEntity()->GetComponent<NNE::Component::Physics::ColliderComponent>();
23
24 if (!collider || !collider->GetShape()) {
25 return;
26 }
27
28 JPH::BodyCreationSettings bodySettings(
29 collider->GetShape(),
30 JPH::RVec3(transform->position.x, transform->position.y, transform->position.z),
31 JPH::Quat::sIdentity(),
32 (mass > 0.0f) ? JPH::EMotionType::Dynamic : (isKinematic ? JPH::EMotionType::Kinematic : JPH::EMotionType::Dynamic),
33 0);
34
35 if (mass > 0.0f) {
36 bodySettings.mMassPropertiesOverride.mMass = mass;
37 }
38
39 JPH::BodyInterface& bodyInterface = physicsSystem->GetBodyInterface();
40 bodyID = bodyInterface.CreateAndAddBody(bodySettings, JPH::EActivation::Activate);

46 if (!bodyID.IsInvalid()) {
47 auto* transform = GetEntity()->GetComponent<NNE::Component::TransformComponent>();

49 JPH::Quat rot = bodyInterface.GetRotation(bodyID);
50 transform->position = glm::vec3(pos.GetX(), pos.GetY(), pos.GetZ());
51 glm::quat glmQuat(rot.GetW(), rot.GetX(), rot.GetY(), rot.GetZ());
