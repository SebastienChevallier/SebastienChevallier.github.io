59void RigidbodyComponent::Awake() {
60 auto physicsSystem = NNE::Systems::PhysicsSystem::GetInstance()->GetPhysicsSystem();
61 auto const* transform = GetEntity()->GetComponent<NNE::Component::TransformComponent>();
62 if (transform) {
63 lastPosition = transform->position;
64 lastRotation = transform->rotation;
65 }
66 auto* collider = GetEntity()->GetComponent<NNE::Component::Physics::ColliderComponent>();
67
68 if (!collider)
69 return;
70
71 if (!collider->GetShape()) {
72 collider->Awake();
73 if (!collider->GetShape())
74 return;
75 }
76
77 if (collider->IsTrigger()) {
78 std::cerr << "Collider on entity '" << GetEntity()->GetName()
79
80 << "' is marked as trigger; forcing IsTrigger to false for physical collisions." << std::endl;
81 collider->SetTrigger(false);
82
83 }
84
85 JPH::EMotionType motionType = isKinematic ? JPH::EMotionType::Kinematic
86 : JPH::EMotionType::Dynamic;
87
88 //JPH::EMotionType motionType = JPH::EMotionType::Dynamic;
89
90
91 JPH::BodyCreationSettings bodySettings(
92 collider->GetShape(),
93 JPH::RVec3(transform->position.x, transform->position.y, transform->position.z),
94 JPH::Quat::sIdentity(),
95 motionType,
96 collider->GetLayer());
97 bodySettings.mIsSensor = collider->IsTrigger();
98
99 JPH::EAllowedDOFs allowed = JPH::EAllowedDOFs::All;
100 if (lockPosition.x) allowed = allowed & ~JPH::EAllowedDOFs::TranslationX;
101 if (lockPosition.y) allowed = allowed & ~JPH::EAllowedDOFs::TranslationY;
102 if (lockPosition.z) allowed = allowed & ~JPH::EAllowedDOFs::TranslationZ;
103 if (lockRotation.x) allowed = allowed & ~JPH::EAllowedDOFs::RotationX;
104 if (lockRotation.y) allowed = allowed & ~JPH::EAllowedDOFs::RotationY;
105 if (lockRotation.z) allowed = allowed & ~JPH::EAllowedDOFs::RotationZ;
106 bodySettings.mAllowedDOFs = allowed;
107
108 if (!isKinematic && mass > 0.0f) {
109 JPH::MassProperties mp = collider->GetShape()->GetMassProperties();
110 mp.ScaleToMass(mass);
111 bodySettings.mMassPropertiesOverride = mp;
112 bodySettings.mOverrideMassProperties = JPH::EOverrideMassProperties::MassAndInertiaProvided;
113 bodySettings.mMotionQuality = JPH::EMotionQuality::LinearCast;
114 }
115
116 if (isKinematic) {
117 bodySettings.mOverrideMassProperties = JPH::EOverrideMassProperties::MassAndInertiaProvided;
118 bodySettings.mMassPropertiesOverride.mMass = 1.0f;
119 bodySettings.mMassPropertiesOverride.mInertia = JPH::Mat44::sIdentity();
120 bodySettings.mGravityFactor = 0.0f;
121 bodySettings.mUseManifoldReduction = false; // Important pour les kinematic
122 bodySettings.mMotionQuality = JPH::EMotionQuality::LinearCast;
123 }
124
125 JPH::BodyInterface& bodyInterface = physicsSystem->GetBodyInterface();
126 bodyID = bodyInterface.CreateAndAddBody(bodySettings, JPH::EActivation::Activate);
127 collider->bodyID = bodyID;
128 NNE::Systems::PhysicsSystem::GetInstance()->RegisterCollider(bodyID, collider);
129}

196void RigidbodyComponent::MoveKinematic(glm::vec3 position, glm::vec3 rotation, float deltaTime) {
197
198 if (!isKinematic)
199 return;
200
201 auto physicsSystem = NNE::Systems::PhysicsSystem::GetInstance()->GetPhysicsSystem();
202 auto& bodyInterface = physicsSystem->GetBodyInterface();
203 if (!bodyID.IsInvalid()) {
204 glm::vec3 radians = glm::radians(rotation);
205 glm::quat q = glm::quat(radians);
206 JPH::Quat orient(q.x, q.y, q.z, q.w);
207 bodyInterface.MoveKinematic(bodyID,
208 JPH::RVec3(position.x, position.y, position.z),
209 orient,
210 deltaTime);
211 }
